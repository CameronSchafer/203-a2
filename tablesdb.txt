CREATE TABLE plants(
ID NUMBER NOT NULL,
featured VARCHAR2(5),
new VARCHAR2(5),
common VARCHAR2(50),
botan VARCHAR2(50),
price NUMBER,
picture VARCHAR2(50),
descrip VARCHAR2(200),
lightlevels VARCHAR2(30),
seasons VARCHAR2(20),
zone VARCHAR2(20),
PRIMARY KEY(ID)
);

CREATE TABLE customers(
ID NUMBER NOT NULL,
username VARCHAR2(50) NOT NULL,
password VARCHAR2(50) NOT NULL,
email VARCHAR2(50) NOT NULL,
phone VARCHAR2(50),
company VARCHAR2(50),
address VARCHAR2(100),
PRIMARY KEY(ID)
);

CREATE TABLE orders(
ID NUMBER NOT NULL,
date_time VARCHAR(20),
customer NUMBER,
paid NUMBER,
PRIMARY KEY(ID),
FOREIGN KEY(customer) REFERENCES CUSTOMERS(ID)
);
